Pour analyser les performances des joueurs de La Liga, nous avons dvelopp une interface RShiny. Cette interface permet de visualiser un graphique qui reprsente les buts marqus des joeurscpar club, tout en offrant la possibilit de filtrer les rsultats en fonction dun seuil dfini.

# Interface utilisateur
ui <- fluidPage(
  titlePanel("Analyse des donnes de la Liga"),
  sidebarLayout(
    sidebarPanel(
      selectInput("club", "Slectionnez un club :", choices = unique(data$Club)),
      numericInput("seuil", "Seuil de buts :", value = 0, min = 0),
      actionButton("update", "Mettre  jour")
    ),
    mainPanel(
      plotOutput("plot_buts")
    )
  )
)

# Serveur
server <- function(input, output) {
  filtered_data <- eventReactive(input$update, {
    data %>%
      filter(Club == input$club, Buts > input$seuil)
  })
  
  output$plot_buts <- renderPlot({
    ggplot(filtered_data()) +
      aes(x = Joueur, y = Buts, fill = Joueur) +
      geom_col() +
      labs(
        title = paste("Nombre de buts pour", input$club),
        x = "Joueur",
        y = "Buts"
      ) +
      theme_minimal() +
      theme(
        axis.text.x = element_text(angle = 45, hjust = 1, size = 10) # Inclinaison des noms
      )
  })
}

#Interface Rshiny
shinyApp(ui = ui, server = server)
